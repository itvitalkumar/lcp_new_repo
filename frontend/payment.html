// ============================================================ // ✅ This page works with ALL revenue-generating features // ============================================================ // ============================================================ // CONFIGURATION // ============================================================ const API_BASE = window.ACTIVE_CONFIG?.API_BASE_URL || 'http://localhost:8000/api'; // ============================================================ // GET URL PARAMETERS // ============================================================ function getParam(name) { const url = new URL(window.location.href); return url.searchParams.get(name); } // --- Required Parameters --- const amount = parseInt(getParam('amount')) || 0; // Amount in rupees const groupId = getParam('group_id') || getParam('groupId') || null; // Group/Job ID const groupType = getParam('type') || getParam('group_type') || 'service'; // Service type // --- Optional Parameters --- const serviceName = getParam('service') || getParam('name') || 'Service Activation'; // Display name const successUrl = getParam('success_url') || 'student_dashboard.html'; // Redirect on success const cancelUrl = getParam('cancel_url') || 'javascript:history.back()'; // Redirect on cancel // ============================================================ // DOM ELEMENTS // ============================================================ const displayAmount = document.getElementById('displayAmount'); const displayService = document.getElementById('displayService'); const detailService = document.getElementById('detailService'); const detailGroupId = document.getElementById('detailGroupId'); const detailStatus = document.getElementById('detailStatus'); const payBtn = document.getElementById('payBtn'); const resultDiv = document.getElementById('paymentResult'); // ============================================================ // RENDER PAGE // ============================================================ function renderPage() { // Display amount and service name displayAmount.innerText = amount || '0'; displayService.innerText = serviceName || '—'; detailService.innerText = serviceName || '—'; detailGroupId.innerText = groupId || 'N/A'; detailStatus.innerText = 'Pending'; detailStatus.style.color = '#e67e22'; // Validate required fields if (!amount || amount <= 0) { payBtn.disabled = true; payBtn.innerHTML = ' Invalid Amount'; showResult('Invalid payment amount. Please go back and try again.', 'error'); } if (!groupId) { payBtn.disabled = true; payBtn.innerHTML = ' Missing Group ID'; showResult('Group ID not found. Please go back and try again.', 'error'); } } // ============================================================ // HELPER FUNCTIONS // ============================================================ function getAuthToken() { return localStorage.getItem('authToken'); } function showResult(message, type) { resultDiv.style.display = 'block'; resultDiv.innerHTML = message; resultDiv.className = type === 'error' ? 'error-msg' : 'success-msg'; } function createConfetti() { const emojis = ['🎉', '✨', '🎊', '🎈', '🌸', '🍾', '🎂']; for (let i = 0; i < 40; i++) { setTimeout(() => { const el = document.createElement('div'); el.textContent = emojis[Math.floor(Math.random() * emojis.length)]; el.style.position = 'fixed'; el.style.left = Math.random() * 100 + '%'; el.style.top = '-20px'; el.style.fontSize = (Math.random() * 15 + 12) + 'px'; el.style.pointerEvents = 'none'; el.style.zIndex = '9999'; el.style.transition = 'transform 2.5s linear, opacity 2.5s linear'; document.body.appendChild(el); requestAnimationFrame(() => { el.style.transform = `translateY(${window.innerHeight + 20}px) rotate(${Math.random() * 720}deg)`; el.style.opacity = '0'; }); setTimeout(() => el.remove(), 2500); }, Math.random() * 300); } } // ============================================================ // LOAD RAZORPAY SCRIPT // ============================================================ function loadRazorpayScript() { return new Promise((resolve, reject) => { if (window.Razorpay) { resolve(true); return; } const script = document.createElement('script'); script.src = 'https://checkout.razorpay.com/v1/checkout.js'; script.onload = () => resolve(true); script.onerror = () => reject(new Error('Failed to load Razorpay SDK')); document.body.appendChild(script); }); } // ============================================================ // INITIATE PAYMENT (CORE LOGIC) // ============================================================ async function initiatePayment() { // --- Validation --- if (!amount || amount <= 0) { alert('Invalid payment amount'); return; } if (!groupId) { alert('Group ID missing. Please go back and try again.'); return; } const token = getAuthToken(); if (!token) { alert('Please login first'); window.location.href = 'login.html'; return; } // --- Show loading state --- payBtn.disabled = true; payBtn.innerHTML = ' Initializing...'; try { // ============================================================ // STEP 1: Create Razorpay Order via Backend // ============================================================ // The backend: // 1. Verifies the group exists and belongs to the user // 2. Creates a Razorpay order // 3. Saves the payment record in the database with status 'pending' // 4. Returns the order_id and key to the frontend // ============================================================ const orderRes = await fetch(`${API_BASE}/payment/create-order`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ amount: amount, group_id: groupId, group_type: groupType }) }); if (!orderRes.ok) { const err = await orderRes.json(); throw new Error(err.detail || 'Order creation failed'); } const orderData = await orderRes.json(); // ============================================================ // STEP 2: Load Razorpay SDK // ============================================================ await loadRazorpayScript(); // ============================================================ // STEP 3: Get current user details // ============================================================ const currentUser = JSON.parse(localStorage.getItem('currentUser') || '{}'); // ============================================================ // STEP 4: Open Razorpay Checkout // ============================================================ // Razorpay will: // 1. Show the payment UI to the user // 2. Collect payment details // 3. Process the payment // 4. Call the handler function with the payment response // ============================================================ const options = { key: orderData.key, amount: orderData.amount * 100, currency: 'INR', name: 'Campus Central', description: `${serviceName} - Activation Fee`, order_id: orderData.order_id, prefill: { name: currentUser.full_name || '', email: currentUser.email || '', contact: currentUser.whatsapp || '' }, theme: { color: '#e67e22' }, handler: async function(response) { // ============================================================ // STEP 5: Verify Payment (Called after successful payment) // ============================================================ // This is the critical step where the backend verifies: // 1. The payment signature (to ensure it's genuine) // 2. The payment record exists in the database // 3. The group is activated // ============================================================ showResult('⏳ Verifying payment...', 'success'); try { const verifyRes = await fetch(`${API_BASE}/payment/verify-payment`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ razorpay_payment_id: response.razorpay_payment_id, razorpay_order_id: response.razorpay_order_id, razorpay_signature: response.razorpay_signature, group_id: groupId, group_type: groupType }) }); const verifyData = await verifyRes.json(); if (verifyData.success) { // ============================================================ // STEP 6: Payment Success — Redirect to success_url // ============================================================ // The backend has: // 1. Verified the signature // 2. Updated the payment status to 'success' // 3. Activated the group (status = 'active') // Now we redirect the user to the dashboard or success page // ============================================================ showResult('✅ Payment successful! Redirecting...', 'success'); createConfetti(); setTimeout(() => { window.location.href = successUrl; }, 2000); } else { throw new Error(verifyData.message || 'Verification failed'); } } catch (err) { showResult(`❌ Payment verification failed: ${err.message}`, 'error'); payBtn.disabled = false; payBtn.innerHTML = ' Retry Payment'; } } }; const razorpay = new Razorpay(options); razorpay.open(); payBtn.disabled = false; payBtn.innerHTML = ' Pay Now'; } catch (err) { showResult(`❌ Error: ${err.message}`, 'error'); payBtn.disabled = false; payBtn.innerHTML = ' Retry Payment'; } } // ============================================================ // INIT — Render the page on load // ============================================================ renderPage(); // Expose function for inline onclick window.initiatePayment = initiatePayment;
